    edge[->, bend right] (waiting)
    edge[<-, bend left]  (semaphore);
  \node[transition] (enter critical) [left=of critical]   {}
    edge[->]             (critical)
    edge[<-,bend left]   (waiting)
    edge[->,bend right]  (semaphore);
\end{tikzpicture}
